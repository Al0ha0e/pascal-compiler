program example(input, output);
    var x, y: integer;
    function gcd(a, b: integer): integer;
        begin 
            a := a * a / b and b * a;
            a := a + b or a - (a - b);
            if b=0 then gcd:=a
            else gcd:=gcd(b, a mod b)
        end;
    begin
        read(x, y);
        write(gcd(x, y))
    end.
