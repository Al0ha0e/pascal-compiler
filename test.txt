program example(input, output);
    var q, w: array[2..1,27..40] of integer;x,y:integer;m,n:real;
    function gcd(var a, b: integer): integer;
        begin 
            x := m mod n;
            y := m div n;
            x := x mod x / y;
            q[a,b] := m;
            w[b,a] := y + 123;
            if b=0 then gcd:=a
            else gcd:=gcd(b, a)
        end;
    begin
        read(q[1,27], y);
        write(gcd(x, y))
    end.
